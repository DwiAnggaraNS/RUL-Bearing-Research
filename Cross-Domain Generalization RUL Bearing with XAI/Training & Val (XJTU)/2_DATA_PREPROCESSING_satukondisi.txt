    "# ============================================================================\n",
    "# PHASE 1 — INTRA-FILE RAW FEATURE EXTRACTION\n",
    "# ============================================================================\n",
    "\n",
    "def extract_features_from_csv(\n",
    "    csv_path: str,\n",
    "    bearing_id: str,\n",
    "    file_index: int,\n",
    ") -> pd.DataFrame:\n",
    "    \"\"\"\n",
    "    Extract 14 raw features from a single per-minute CSV file.\n",
    "\n",
    "    The file is sliced into overlapping 1024-sample windows with 512-sample stride.\n",
    "    Features are extracted per window from the horizontal (H) vibration channel.\n",
    "    Outputs exactly 63 rows/segments per CSV file.\n",
    "\n",
    "    Args:\n",
    "        csv_path:   Absolute path to the per-minute CSV file.\n",
    "        bearing_id: Canonical bearing identifier (e.g., \"Bearing1_3\").\n",
    "        file_index: 1-based integer index of the CSV file in the bearing's\n",
    "                    chronological sequence.\n",
    "\n",
    "    Returns:\n",
    "        DataFrame of shape (63, 14+meta) or None on failure.\n",
    "    \"\"\"\n",
    "    try:\n",
    "        df_raw = pd.read_csv(csv_path, header=None)\n",
    "    except Exception as exc:\n",
    "        print(f\"    [WARNING] Cannot read {csv_path}: {exc}\")\n",
    "        return None\n",
    "\n",
    "    # XJTU-SY format: column 0 = horizontal, column 1 = vertical.\n",
    "    if df_raw.shape[1] < 1:\n",
    "        return None\n",
    "\n",
    "    # Coerce to numeric to safely ignore header strings\n",
    "    h_series = pd.to_numeric(df_raw.iloc[:, 0], errors='coerce')\n",
    "    h_signal = h_series.dropna().to_numpy(dtype=np.float64)\n",
    "    n_total   = len(h_signal)\n",
    "\n",
    "    # For 32768 samples, window=1024, stride=512 -> exactly 63 segments\n",
    "    if n_total < SEGMENT_LENGTH:\n",
    "        return None\n",
    "\n",
    "    extractor = CrossDomainFeatureExtractor(sampling_rate=XJTU_SAMPLING_RATE)\n",
    "\n",
    "    # Use stride of 512 (which is SEGMENT_LENGTH // 2)\n",
    "    stride = 512\n",
    "    n_windows = (n_total - SEGMENT_LENGTH) // stride + 1\n",
    "    \n",
    "    # Cap to exactly 63 to avoid uneven sizes if file has extra samples\n",
    "    n_windows = min(n_windows, 63)\n",
    "\n",
    "    rows = []\n",
    "    for w in range(n_windows):\n",
    "        start = w * stride\n",
    "        seg   = h_signal[start: start + SEGMENT_LENGTH]\n",
    "        feats = extractor.extract_all_features(seg)\n",
    "        feats[\"Bearing_ID\"]  = bearing_id\n",
    "        feats[\"File_Index\"]  = file_index\n",
    "        feats[\"Segment_Index\"] = w\n",
    "        rows.append(feats)\n",
    "\n",
    "    if not rows:\n",
    "        return None\n",
    "\n",
    "    return pd.DataFrame(rows)"

    "# ============================================================================\n",
    "# PHASE 3 — CAUSAL GROUND TRUTH CONSTRUCTION\n",
    "# ============================================================================\n",
    "\n",
    "def construct_causal_hi(df_segments: pd.DataFrame) -> Tuple[pd.DataFrame, dict]:\n",
    "    \"\"\"\n",
    "    Construct piecewise-linear Health Index (HI) targets on the 63-segment/min series.\n",
    "\n",
    "    Execution order is strictly causal:\n",
    "    1. CUSUM is applied to the smoothed RMS column.\n",
    "    2. HI target is derived from the detected change point (piecewise linear).\n",
    "    3. HI target is smoothed with EMA.\n",
    "    4. Isotonic Regression is applied to force strict monotonically decreasing labels.\n",
    "\n",
    "    Args:\n",
    "        df_segments: Segment-level DataFrame with RAW_FEATURE_COLS columns.\n",
    "\n",
    "    Returns:\n",
    "        Tuple of (df with hi_target + cusum_change_point columns, metadata dict).\n",
    "    \"\"\"\n",
    "    df_out      = df_segments.copy()\n",
    "    n_total     = len(df_out)\n",
    "    bearing_id  = df_out[\"Bearing_ID\"].iloc[0]\n",
    "\n",
    "    rms_col = \"td_rms\"\n",
    "    if rms_col not in df_out.columns:\n",
    "        raise ValueError(f\"Column '{rms_col}' not found in {bearing_id}.\")\n",
    "\n",
    "    # 1. Smooth RMS using rolling window (approx 1 minute = 63 segments)\n",
    "    smoothed_rms = (\n",
    "        df_out[rms_col]\n",
    "        .rolling(window=CUSUM_SMOOTHING_WIN, min_periods=1, center=True)\n",
    "        .mean()\n",
    "        .values\n",
    "    )\n",
    "\n",
    "    # 2. CUSUM\n",
    "    detector = UnivariateCUSUMDetector(\n",
    "        baseline_ratio=CUSUM_BASELINE_RATIO,\n",
    "        k_factor=CUSUM_K_FACTOR,\n",
    "        h_factor=CUSUM_H_FACTOR,\n",
    "    )\n",
    "    change_point, _ = detector.fit_predict(smoothed_rms)\n",
    "\n",
    "    # 3. Piecewise linear HI\n",
    "    hi = np.ones(n_total, dtype=np.float64)\n",
    "    degradation_len = n_total - change_point\n",
    "    if degradation_len > 1:\n",
    "        hi[change_point:] = np.linspace(1.0, 0.0, degradation_len)\n",
    "    elif degradation_len == 1:\n",
    "        hi[change_point] = 0.0\n",
    "        \n",
    "    # 4. Smooth HI with EMA\n",
    "    hi_ema = pd.Series(hi).ewm(alpha=HI_EMA_ALPHA, adjust=False).mean().values\n",
    "    \n",
    "    # 5. Monotonicity via Isotonic Regression (increasing=False)\n",
    "    iso_reg = IsotonicRegression(increasing=False, out_of_bounds='clip')\n",
    "    hi_iso = iso_reg.fit_transform(np.arange(n_total), hi_ema)\n",
    "\n",
    "    df_out[\"hi_target\"] = hi_iso\n",
    "    df_out[\"cusum_change_point\"] = change_point\n",
    "\n",
    "    meta = {\n",
    "        \"Bearing_ID\": bearing_id,\n",
    "        \"Total_Segments\": n_total,\n",
    "        \"Total_Minutes\": n_total / 63.0,\n",
    "        \"Change_Point_Idx\": change_point,\n",
    "        \"Change_Point_Min\": change_point / 63.0,\n",
    "        \"Degradation_Len\": degradation_len,\n",
    "    }\n",
    "    return df_out, meta"
